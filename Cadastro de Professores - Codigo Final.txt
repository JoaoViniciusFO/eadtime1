    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:

        if (JTabela.getSelectedRow() !=-1){

            JTabela.setValueAt(txtNome.getText(), JTabela.getSelectedRow(), 0);
            JTabela.setValueAt(txtData.getText(), JTabela.getSelectedRow(), 1);
            JTabela.setValueAt(txtFixo.getText(), JTabela.getSelectedRow(), 2);
            JTabela.setValueAt(txtCelular.getText(), JTabela.getSelectedRow(), 3);
            JTabela.setValueAt(txtEmail.getText(), JTabela.getSelectedRow(), 4);
            JTabela.setValueAt(txtCPF.getText(), JTabela.getSelectedRow(), 5);
        }
    }                                        

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        if (JTabela.getSelectedRow()!=-1) {
            DefaultTableModel Valor = (DefaultTableModel) JTabela.getModel();
            Valor.removeRow(JTabela.getSelectedRow());
        }else {
            JOptionPane.showMessageDialog(null, "Selecione o Professor para Excluir!");

        }
    }                                        

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         

        String nome = txtNome.getText().trim();
        String data = txtData.getText().trim();
        String fixo = txtFixo.getText().trim();
        String celular = txtCelular.getText().trim();
        String email = txtEmail.getText().trim();
        String CPF = txtCPF.getText().trim();

        DefaultTableModel Valor = (DefaultTableModel) JTabela.getModel();
        Valor.addRow(new String[]{nome, data, fixo, celular, email, CPF});

        txtNome.setText("");
        txtData.setText("");
        txtFixo.setText("");
        txtCelular.setText("");
        txtEmail.setText("");
        txtCPF.setText("");
    }
