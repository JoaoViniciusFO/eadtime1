    private void jButtonCadastrarActionPerformed(java.awt.event.ActionEvent evt) {                                                 
        // TODO add your handling code here:
        String periodo = jComboBoxPeriodo.getSelectedItem().toString().trim();
        String tempo = jComboBoxTempo.getSelectedItem().toString().trim();
        String tipocurso = jComboBoxTipoCurso.getSelectedItem().toString().trim();
        String curso= jComboBoxCurso.getSelectedItem().toString().trim();
        String disciplina = jComboBoxDisciplina.getSelectedItem().toString().trim();
        String cargahoraria = jTextFieldCargaHoraria.getText().trim();
        String datainicio = jTextFieldDataInicio.getText().trim();
        String datafim = jTextFieldDataFim.getText().trim();
        String descricao = jTextAreaDescricao.getText().trim();

        DefaultTableModel val = (DefaultTableModel) jTableCadastroDisciplinas.getModel();

        val.addRow(new String[]{periodo, tempo, tipocurso, curso, disciplina, cargahoraria, datainicio, datafim, descricao});

        jComboBoxPeriodo.setSelectedItem("");
        jComboBoxTempo.setSelectedItem("");
        jComboBoxTipoCurso.setSelectedItem("");
        jComboBoxCurso.setSelectedItem("");
        jComboBoxDisciplina.setSelectedItem("");
        jTextFieldCargaHoraria.setText("");
        jTextFieldDataInicio.setText("");
        jTextFieldDataFim.setText("");
        jTextAreaDescricao.setText("");
            
    }

    private void jButtonAlterarActionPerformed(java.awt.event.ActionEvent evt) {                                               
        // TODO add your handling code here:
        
                if(jTableCadastroDisciplinas.getSelectedRow() != -1 ){

            // JTableCadastro.setValueAt(Idade, ERROR, NORMAL);
            jTableCadastroDisciplinas.setValueAt(jComboBoxPeriodo.getSelectedItem(), jTableCadastroDisciplinas.getSelectedRow(), 0);
            jTableCadastroDisciplinas.setValueAt(jComboBoxTempo.getSelectedItem(), jTableCadastroDisciplinas.getSelectedRow(), 1);
            jTableCadastroDisciplinas.setValueAt(jComboBoxTipoCurso.getSelectedItem(), jTableCadastroDisciplinas.getSelectedRow(), 2);
            jTableCadastroDisciplinas.setValueAt(jComboBoxCurso.getSelectedItem(), jTableCadastroDisciplinas.getSelectedRow(), 3);
            jTableCadastroDisciplinas.setValueAt(jComboBoxDisciplina.getSelectedItem(), jTableCadastroDisciplinas.getSelectedRow(), 4);
            jTableCadastroDisciplinas.setValueAt(jTextFieldCargaHoraria.getText(), jTableCadastroDisciplinas.getSelectedRow(),5);
            jTableCadastroDisciplinas.setValueAt(jTextFieldDataInicio.getText(), jTableCadastroDisciplinas.getSelectedRow(), 6);
            jTableCadastroDisciplinas.setValueAt(jTextFieldDataFim.getText(), jTableCadastroDisciplinas.getSelectedRow(), 7);
            jTableCadastroDisciplinas.setValueAt(jTextAreaDescricao.getText(), jTableCadastroDisciplinas.getSelectedRow(), 8);
            
        }

    }                                              

    private void jButtonExcluirActionPerformed(java.awt.event.ActionEvent evt) {                                               
        // TODO add your handling code here:
        
    if(jTableCadastroDisciplinas.getSelectedRow()!= -1){

    DefaultTableModel dtmCadastroDisciplinas = (DefaultTableModel) jTableCadastroDisciplinas.getModel();
    dtmCadastroDisciplinas.removeRow(jTableCadastroDisciplinas.getSelectedRow());
        }else{
            JOptionPane.showMessageDialog(null, "Selecione uma linha para excluir!");
        }
    }
